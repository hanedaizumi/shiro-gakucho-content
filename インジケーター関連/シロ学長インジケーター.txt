npr_sq(source) => math.pow(source, 2)
npr_gaussian(source, bandwidth) => math.exp(-npr_sq(source / bandwidth) / 2) / math.sqrt(2 * math.pi)
npr_laplace(source, bandwidth) => (1 / (2 * bandwidth)) * math.exp(-math.abs(source/bandwidth))
npr_epanechnikov(source, bandwidth) => math.abs(source/bandwidth) <= 1 ? (3/4.) * (1 - npr_sq(source/bandwidth)) : 0.0

npr_kernel_calc(source, bandwidth, style) =>
    switch style
        "Gaussian"      => npr_gaussian(source, bandwidth)
        "Epanechnikov"  => npr_epanechnikov(source, bandwidth)
        => npr_laplace(source, bandwidth)

npr_precalculate(bandwidth, kernel) =>
    var float[] weights = array.new_float(0)
    var float sumw = 0.0
    if barstate.isfirst
        weights := array.new_float(bandwidth)
        for i = 0 to bandwidth - 1
            j = math.pow(i, 2) / math.pow(bandwidth, 2)
            weight = npr_kernel_calc(j, 1, kernel)
            array.set(weights, i, weight)
            sumw += weight
        [weights, sumw]
    else
        [weights, sumw]

npr_non_repaint_stdev(source, bandwidth, deviations, kernel) =>
    var float npr_sum = na
    var float npr_stdev = na
    var color band_color = na
    [weights, sumw] = npr_precalculate(bandwidth, kernel)
    float sum = 0.0
    float sumsq = 0.0
    for i = 0 to bandwidth - 1
        weight = array.get(weights, i)
        sum += nz(source[i]) * weight
    npr_sum := sum / sumw
    npr_direction = npr_sum > nz(npr_sum[1])
    band_color := npr_direction ? color.red : color.green
    for i = 0 to bandwidth - 1
        sumsq += math.pow(source[i] - npr_sum, 2)
    npr_stdev := math.sqrt(sumsq / (bandwidth - 1)) * deviations
    npr_upper = npr_sum + npr_stdev
    npr_lower = npr_sum - npr_stdev
    npr_upper_slope = (npr_upper - npr_upper[3]) / 3
    npr_lower_slope = (npr_lower - npr_lower[3]) / 3
    [npr_sum, npr_stdev, band_color, npr_upper, npr_lower, npr_upper_slope, npr_lower_slope]

calcSuperTrend(atrLen, multiplier) =>
    atrValue = ta.atr(atrLen)
    hl2Val = (high + low) / 2
    basic_ub = hl2Val + (atrValue * multiplier)
    basic_lb = hl2Val - (atrValue * multiplier)
    var float final_ub = na
    var float final_lb = na
    final_ub := na(final_ub[1]) ? basic_ub : (basic_ub < final_ub[1] or close[1] > final_ub[1]) ? basic_ub : final_ub[1]
    final_lb := na(final_lb[1]) ? basic_lb : (basic_lb > final_lb[1] or close[1] < final_lb[1]) ? basic_lb : final_lb[1]
    var float st = na
    if na(st[1])
        st := final_ub
    else
        if st[1] == final_ub[1]
            st := close <= final_ub ? final_ub : final_lb
        else if st[1] == final_lb[1]
            st := close >= final_lb ? final_lb : final_ub
        else
            st := close <= final_ub ? final_ub : final_lb
    trend_dir = st == final_lb ? 1 : -1
    [st, trend_dir]
